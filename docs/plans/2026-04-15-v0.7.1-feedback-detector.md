# hooks/lib/evidence-extract.sh
# Evidence extraction for feedback detector (v0.7.1).
# Two pure functions; no side effects. Source after $MEMORY_DIR is available.
#
#   evidence_from_frontmatter <path> → stdout: newline-separated phrases
#   evidence_from_body        <path> → stdout: newline-separated unique tokens

# ---------- evidence_from_frontmatter ----------
# Reads YAML array under `evidence:` key in frontmatter (between `---` fences).
# Accepts both inline-string list items and multi-line block form.
# Output: one phrase per line, unquoted, trimmed. Empty if field absent.
evidence_from_frontmatter() {
  local file="$1"
  [ -f "$file" ] || return 0
  awk '
    BEGIN { in_fm = 0; in_evid = 0 }
    /^---[[:space:]]*$/ {
      if (in_fm == 0) { in_fm = 1; next }
      else { exit }
    }
    in_fm == 0 { next }
    /^evidence:[[:space:]]*$/ { in_evid = 1; next }
    in_evid == 1 {
      # Stop when a new top-level key starts (no leading whitespace, ends with colon)
      if (/^[^[:space:]-].*:[[:space:]]*$/ || /^[^[:space:]].*:[[:space:]]/) {
        in_evid = 0; next
      }
      # Match list item: leading dash, then optional quoted or unquoted value
      if (match($0, /^[[:space:]]*-[[:space:]]+(.+)$/, arr)) {
        val = arr[1]
        # Strip surrounding quotes
        gsub(/^"|"$/, "", val)
        gsub(/^'"'"'|'"'"'$/, "", val)
        # Trim whitespace
        gsub(/^[[:space:]]+|[[:space:]]+$/, "", val)
        if (val != "") print val
      }
    }
  ' "$file" 2>/dev/null
}
```

```bash
# ---------- Stop-list (ru + en + markdown noise) ----------
# Built-in, case-lowercased. Tokens below minimum length (4) are also filtered.
_EVIDENCE_STOPWORDS="
это что как для при если или можно нужно быть очень только также
этот эта эти они она оно свои свой такой такая такие была были есть
чего чтобы потому поэтому вообще просто всегда когда даже тоже уже
никогда всегда ничего всего нельзя надо между через чем тем того
the this that with when from have will been into than then also
what where which while would could should about their there these
those such more some only many most just even much very your you
does done same here both each whether against because before after
about across around behind beyond therefore however moreover
why how note example apply reason always never must should
body text content file path memory line code block
"

# ---------- evidence_from_body ----------
# Reads body (content after closing frontmatter `---`), excluding:
#   - Lines starting with **Why:** or **How to apply:**
#   - Fenced code blocks (between triple backticks)
#   - Blockquote lines (`>`)
#   - The slug (filename stem) itself — filtered post-tokenization
# Tokenization: lowercase, split on non-alphanumeric, filter len<4 and stop-words.
# Output: unique tokens, one per line.
# Truncates input at 10 KB to bound work.
evidence_from_body() {
  local file="$1"
  [ -f "$file" ] || return 0
  local slug
  slug=$(basename "$file" .md | tr '[:upper:]' '[:lower:]')

  # Build regex-safe stop-list as a pipe-separated awk pattern
  local stopwords="$_EVIDENCE_STOPWORDS"

  head -c 10240 "$file" 2>/dev/null | awk -v slug="$slug" -v stop="$stopwords" '
    BEGIN {
      # Build stop-set map
      n = split(stop, a, /[[:space:]]+/)
      for (i = 1; i <= n; i++) if (a[i] != "") stopset[a[i]] = 1
      stopset[slug] = 1
      in_fm = 0; fm_closed = 0; in_code = 0
    }
    /^---[[:space:]]*$/ {
      if (fm_closed == 0 && in_fm == 0) { in_fm = 1; next }
      if (fm_closed == 0 && in_fm == 1) { in_fm = 0; fm_closed = 1; next }
    }
    in_fm == 1 { next }
    fm_closed == 0 { next }
    /^```/ { in_code = 1 - in_code; next }
    in_code == 1 { next }
    /^[[:space:]]*>/ { next }
    /^[[:space:]]*\*\*Why:\*\*/ { next }
    /^[[:space:]]*\*\*How to apply:\*\*/ { next }
    {
      # Lowercase whole line
      line = tolower($0)
      # Replace non-alphanumeric (incl. Cyrillic? — BWK awk is byte-oriented;
      # Cyrillic letters are multi-byte UTF-8 and will be split. Since we only
      # need ascii-lowercase tokens in practice, keep it simple and let
      # Cyrillic tokens degrade to byte fragments — they will fail len<4 filter
      # or simply be skipped.)
      gsub(/[^a-z0-9_]+/, " ", line)
      n = split(line, tok, /[[:space:]]+/)
      for (i = 1; i <= n; i++) {
        t = tok[i]
        if (length(t) < 4) continue
        if (t in stopset) continue
        if (!(t in seen)) { seen[t] = 1; print t }
      }
    }
  ' 2>/dev/null
}
```

Note on Cyrillic: BWK awk on macOS treats bytes, so `gsub(/[^a-z0-9_]+/, ...)` won't properly tokenize Russian. For v0.7.1, accept this limitation — most evidence (code patterns, English concepts) IS ASCII. Cyrillic-heavy memories should use explicit `evidence:` frontmatter. Document this in the CHANGELOG.

# --- Feedback loop: useful/silent signal for injected memories ---
# Uses hooks/lib/evidence-extract.sh to match memory content against assistant text.
# evidence: frontmatter path (threshold ≥1) or body-mining fallback (threshold ≥2).
if [ -f "$WAL_FILE" ] && [ -n "$SAFE_SESSION_ID" ] && [ -n "$TRANSCRIPT_PATH" ] && [ -f "$TRANSCRIPT_PATH" ]; then
  _INJECTED_ALL=$(awk -F'|' -v sid="$SAFE_SESSION_ID" '
    $4 == sid && ($2 == "inject" || $2 == "trigger-match") { print $3 }
  ' "$WAL_FILE" 2>/dev/null | sort -u)

  if [ -n "$_INJECTED_ALL" ]; then
    # Extract assistant text only — skip tool calls and user messages.
    _ASSIST_TEXT=$(awk '
      /"type":"assistant"/ && /"text":/ {
        t = $0
        while (match(t, /"text":"[^"]*"/)) {
          print substr(t, RSTART+8, RLENGTH-9)
          t = substr(t, RSTART+RLENGTH)
        }
      }
    ' "$TRANSCRIPT_PATH" 2>/dev/null)

    # Lowercase once
    _ASSIST_TEXT_LC=$(printf '%s' "$_ASSIST_TEXT" | tr '[:upper:]' '[:lower:]')

    # shellcheck source=lib/evidence-extract.sh
    . "$(dirname "$0")/lib/evidence-extract.sh" 2>/dev/null || true
    # shellcheck source=lib/wal-lock.sh
    . "$(dirname "$0")/lib/wal-lock.sh" 2>/dev/null || true

    _FEEDBACK_LINES=""
    _DIAG_LINES=""
    while IFS= read -r _inj_slug; do
      [ -z "$_inj_slug" ] && continue

      _MEM_FILE=$(find "$MEMORY_DIR" -maxdepth 3 -name "${_inj_slug}.md" 2>/dev/null | head -1)
      if [ -z "$_MEM_FILE" ]; then
        _DIAG_LINES="${_DIAG_LINES}${TODAY}|evidence-missing|${_inj_slug}|${SAFE_SESSION_ID}
"
        _FEEDBACK_LINES="${_FEEDBACK_LINES}${TODAY}|trigger-silent|${_inj_slug}|${SAFE_SESSION_ID}
"
        continue
      fi

      _EVID=$(evidence_from_frontmatter "$_MEM_FILE" 2>/dev/null)
      if [ -n "$_EVID" ]; then
        _THRESHOLD=1
      else
        _EVID=$(evidence_from_body "$_MEM_FILE" 2>/dev/null)
        if [ -z "$_EVID" ]; then
          _DIAG_LINES="${_DIAG_LINES}${TODAY}|evidence-empty|${_inj_slug}|${SAFE_SESSION_ID}
"
          _FEEDBACK_LINES="${_FEEDBACK_LINES}${TODAY}|trigger-silent|${_inj_slug}|${SAFE_SESSION_ID}
"
          continue
        fi
        _THRESHOLD=2
      fi

      # Count unique hits (case-insensitive substring / word-boundary)
      _HITS=0
      _SEEN=""
      while IFS= read -r _phrase; do
        [ -z "$_phrase" ] && continue
        _phrase_lc=$(printf '%s' "$_phrase" | tr '[:upper:]' '[:lower:]')
        # Dedup identical phrases
        case "|$_SEEN|" in *"|$_phrase_lc|"*) continue ;; esac
        if printf '%s' "$_ASSIST_TEXT_LC" | grep -qF "$_phrase_lc" 2>/dev/null; then
          _HITS=$((_HITS + 1))
          _SEEN="${_SEEN}|${_phrase_lc}"
        fi
      done <<< "$_EVID"

      if [ "$_HITS" -ge "$_THRESHOLD" ]; then
        _FEEDBACK_LINES="${_FEEDBACK_LINES}${TODAY}|trigger-useful|${_inj_slug}|${SAFE_SESSION_ID}
"
      else
        _FEEDBACK_LINES="${_FEEDBACK_LINES}${TODAY}|trigger-silent|${_inj_slug}|${SAFE_SESSION_ID}
"
      fi
    done <<< "$_INJECTED_ALL"

    if [ -n "$_FEEDBACK_LINES" ] || [ -n "$_DIAG_LINES" ]; then
      wal_run_locked bash -c 'printf "%s%s" "$1" "$2" >> "$3"' _ \
        "$_FEEDBACK_LINES" "$_DIAG_LINES" "$WAL_FILE" 2>/dev/null || true
    fi
  fi
fi
```

## [0.7.1] - 2026-04-15

Feedback detector rewrite — fixes the 0/1472 useful:inject ratio discovered during v0.7 dogfooding.

### Fixed

- **`session-stop.sh` feedback detector** — previously did literal `grep -F "$slug"` against assistant text, which never matches because the assistant applies rules silently (you fix tz-aware datetime, not write "code-approach"). Replaced with content-based matching via new `hooks/lib/evidence-extract.sh`.

### Added

- **`hooks/lib/evidence-extract.sh`** — two pure functions:
  - `evidence_from_frontmatter` — parses optional `evidence:` YAML array from memory frontmatter. Threshold: ≥1 phrase matched (case-insensitive substring) for `trigger-useful`.
  - `evidence_from_body` — fallback. Extracts unique content tokens from memory body (excludes frontmatter, `**Why:**`/`**How to apply:**` sections, code blocks, blockquotes, the slug itself). Filters by length ≥4 and built-in ~50-word ru+en stop-list. Threshold: ≥2 unique tokens matched.
- **`evidence:` frontmatter field** — optional, documented in README. Authors can specify exact phrases that indicate application.
- **New WAL events:**
  - `evidence-missing|slug|session` — injected memory file not found at session-stop. Useful to detect broken MEMORY.md regen or deleted-mid-session.
  - `evidence-empty|slug|session` — memory has no evidence frontmatter AND body-mining yields zero tokens. Signal that the memory is too short/frontmatter-heavy to be detectable.

### Signal bias

Detector favors `trigger-useful` (low threshold). Rationale: downstream auto-disable of noisy triggers must avoid penalizing working memories. False `silent` is more costly than false `useful`.

### Limitations

- Body-mining is ASCII-oriented (macOS BWK awk treats Cyrillic as bytes). Cyrillic-heavy memories should use explicit `evidence:` frontmatter.
- Multi-rule memories (e.g. `code-approach` with 5 unrelated rules) may always register useful if ANY rule's vocabulary overlaps. If auto-disable later surfaces this, the memory is a candidate for splitting — good practice independently.
